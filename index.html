快速使用

1.引入文件

<script src="js/jquery-1.11.1.min.js"></script>
<link rel="stylesheet" href="css/QFocus.css" />
<script src="js/QFocus.js"></script>

2.HTML结构

<div id="focus1" class="QFocus">
    <ul>
        <li><a href="#"><img src="images/1.jpg" /></a></li>
        <li><a href="#"><img src="images/2.jpg" /></a></li>
        <li><a href="#"><img src="images/3.jpg" /></a></li>
        <li><a href="#"><img src="images/4.jpg" /></a></li>
    </ul>
</div>

3.插件调用

$("#focus1").QFocus();

选项

•  type

描述
切换效果类型
类型
String
可选值
"slide""fade"
默认值
"slide"

示例

$("#focus1").QFocus({
    type:"fade"
});

•  title

描述
是否显示标题
类型
Boolean
默认值
false
提示
设为true时，<a>标签需要有title属性

示例

<div id="focus1" class="QFocus">
    <ul>
        <li><a href="#" title="标题一"><img src="images/1.jpg" /></a></li>
        <li><a href="#" title="标题二"><img src="images/2.jpg" /></a></li>
        <li><a href="#" title="标题三"><img src="images/3.jpg" /></a></li>
        <li><a href="#" title="标题四"><img src="images/4.jpg" /></a></li>
    </ul>
</div>
$("#focus1").QFocus({
    title:true
});

•  navControl

描述
是否显示切换条
类型
Boolean
默认值
true

示例

$("#focus1").QFocus({
    navControl:false
});

•  arrowControl

描述
是否显示左右切换按钮
类型
Boolean
默认值
true

示例

$("#focus1").QFocus({
    arrowControl:false
});

•  自定义切换按钮

描述
可自定义左右切换按钮
前置条件
arrowControl需为true

示例

<span id="prev" class="btn btn-prev"><</span>
<span id="next" class="btn btn-next">></span>
$("#focus1").QFocus({
    arrowControl: true,
    prevBtn: $("#prev"),
    nextBtn: $("#next")
});
< >

•  afterInit()

描述
初始化结束后的回调函数
类型
Function

示例

$("#focus1").QFocus({
    afterInit: function() {
        alert("初始化完成了呢！");
    }
});

•  width

描述
设置宽度
类型
Number/String
默认值
600
说明
与jQuery的width()一致

示例

$("#focus1").QFocus({
    width: "100%"
});

•  height

描述
设置高度
类型
Number/String
默认值
300
说明
与jQuery的height()一致

示例

$("#focus1").QFocus({
    height: 400
});

•  interval

描述
设置自动播放时的时间间隔
类型
Number
单位
毫秒(ms)
默认值
5000

示例

$("#focus1").QFocus({
    interval: 3000
});

•  duration

描述
设置每次切换动画的持续时间
类型
Number
单位
毫秒(ms)
默认值
500

示例

$("#focus1").QFocus({
    duration: 800
});
